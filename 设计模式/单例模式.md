# 单例模式

* 实例如果已经创建，则直接返回
* 实现可以修改单例
* 惰性单例

```js
const Alert = (
  function() {
    let instance
    function Alert (content) {
      // 如果实例是空的就指向this
      console.log(instance)
      if (!instance) {
        instance = this
      }
      instance.init(content)
      return instance
    }
    Alert.prototype.init = function(content) {
      this.content = content
    }
    return Alert
  }
)()

const a = new Alert('a')
console.log(a.content)
const b = new Alert('b')
console.log(b.content)
console.log(a === b) // true
```
